```sh
# ================================================
#      Check Vulnerabilities - searchsploit
# ================================================

# for searchsploit
grep -E '^[0-9]+' nmap_open_services.txt \
  | cut -d' ' -f6- \
  | sed 's/  */ /g' \
  | sed '/^$/d' \
  | sort -u \
  | while read -r line; do
    trimmed=$(echo "$line" | xargs) 
    if [[ -n "$trimmed" ]]; then
        echo "searchsploit \"$trimmed\"" >> searchsploit.sh
    fi
done
chmod +x searchsploit.sh; ./searchsploit.sh
